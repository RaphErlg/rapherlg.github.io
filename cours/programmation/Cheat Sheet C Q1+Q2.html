⚡ Opérateurs

Arith   : + - * /  %        (% = modulo)
Comp    : == != < > <= >=
Logique : && || !
Bit     : & | ^ ~ << >>
Compose : += -= *= /= %= |= &=
Ternaire: cond ? si_vrai : si_faux
sizeof  : sizeof(x), sizeof(tab)/sizeof(tab[0])

Priorités (haut → bas) : ! ++ -- → * / % → + - → < > <= >= → == != → & → ^ → | → && → || → ?: → =

Cast : (double)a / b

🔁 Contrôle de flux

// if / else
if (x == 5)      {...}
else if (x > 5)  {...}
else             {...}

// switch
switch (c) {
    case 'A':
    case 'B':       // fall-through OK
        ...
        break;
    default:
        ...;
}

// for
for (int i = 0; i < n; i++) {...}

// while
while (cond) {...}
do {...} while (cond);

// break  → sort de la boucle / switch
// continue → itération suivante
// return → sort de la fonction

typedef enum salle Salle;
enum salle {
    CHAMBRE = 0b0001,
    SALON   = 0b0010,
    SDB     = 0b0100,
    CUISINE = 0b1000
};

// 4 manipulations
int c = SALON | CHAMBRE;            // ALLUMER (combiner) : |
if ((c & SALON) != 0) ...           // TESTER : & + != 0 (parenthèses !)
c = c & ~SALON;                     // ÉTEINDRE : & ~
c = c ^ SALON;                      // BASCULER : ^

// Fonction de test
bool estAllumee(int conf, Salle s) {
    return (conf & s) != 0;
}

// Masque (isoler plusieurs bits)

🎁 Patterns essentiels

Validation d'entrée

int valide(int min, int max) {
    int n;
    scanf_s("%d", &n);
    while (n < min || n > max) {
        printf("Recommencez : ");
        scanf_s("%d", &n);
    }
    return n;
}

Recherche linéaire (sans break)

int indice(int cle, T tab[], int n) {
    int i = 0;
    while (i < n && tab[i].code != cle) i++;
    return i;       // i == n si pas trouvé
}

Tous identiques ?

bool tousIdentiques(char tab[], int n) {
    int i = 1;
    while (i < n && tab[i] == tab[0]) i++;
    return (i == n);
}

Compter occurrences

int nbOcc(int tab[], int n, int v) {
    int c = 0;
    for (int i = 0; i < n; i++)
        if (tab[i] == v) c++;
    return c;
}

Indice du max

int iMax = 0;
for (int i = 1; i < n; i++)
    if (tab[i] > tab[iMax]) iMax = i;

⚠️ Pièges récurrents

// Syntaxe
if (x = 5)            // ❌  affectation au lieu de comparaison
#define PI = 3.14     // ❌  pas de = dans #define
scanf("%d", x)        // ❌  manque &
struct s { int a; }   // ❌  manque ;

// Types
(a + b) / 3           // ⚠️  division entière si a, b sont int
%d pour un double     // ❌
nom == "Alice"        // ❌  compare des adresses
nom = "Alice"         // ❌  utiliser strcpy

// Logique
while (c != 'A' || c != 'B')        // ❌  toujours vrai → && au lieu de ||
for (i=0; i <= N; i++) tab[i]       // ❌  i = N hors limite
tab[mois]                            // ❌  si mois ∈ [1..12] → tab[mois - 1]

// Bits
if (config & SALON != 0)            // ❌  parenthèses
if (config && SALON)                // ❌  && au lieu de &

// Fichiers
fopen sans vérifier NULL            // ❌
oublier fclose                       // ❌  données perdues
feof sans lecture avant la boucle   // ⚠️  dernière valeur dupliquée

// Pointeurs
f(&tab)                              // ❌  tableau = déjà adresse
*p++ vs (*p)++                       // ⚠️  toujours mettre les ()
