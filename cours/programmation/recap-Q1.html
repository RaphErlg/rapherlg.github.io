Les 4 cas

// 1. Sans entrée, sans retour (procédure)
void afficherMenu(void) {
    printf("Menu\n");
}
afficherMenu();

// 2. Avec entrée, sans retour
void afficherDouble(int n) {
    printf("%d\n", n * 2);
}
afficherDouble(5);

// 3. Avec entrée, avec retour
int cube(int n) {
    return n * n * n;
}
int r = cube(3);    // r = 27

// 4. Sans entrée, avec retour
int lireEntier(void) {
    int n;
    scanf("%d", &n);
    return n;
}
int val = lireEntier();

📦 Tableaux

// Déclaration
#define TAILLE 5
int tab[TAILLE];

// Déclaration + initialisation
int tab[5] = {10, 20, 30, 40, 50};
int tab[]  = {10, 20, 30, 40, 50};   // taille déduite automatiquement
int tab[5] = {0};                    // tout initialisé à 0

// Accès (indices de 0 à TAILLE-1 !)
tab[0] = 99;
printf("%d", tab[2]);

// Parcours complet
for (int i = 0 ; i < TAILLE ; i++) {
    printf("%d ", tab[i]);
}

// Tableau en paramètre de fonction (passage par ADRESSE)
// → la fonction a accès à la mémoire originale du tableau

// Fonction qui LIT → ne modifie pas le tableau
void afficher(int tab[], int taille) {
    for (int i = 0 ; i < taille ; i++)
        printf("%d ", tab[i]);
}

// Fonction qui ÉCRIT → modifie le tableau original !
void remplir(int tab[], int taille) {
    for (int i = 0 ; i < taille ; i++)
        scanf("%d", &tab[i]);
}

// Appels
afficher(tab, TAILLE);   // lecture seule, original inchangé
remplir(tab, TAILLE);    // modifie le tableau original !
